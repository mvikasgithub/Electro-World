(
	id IDENTITY,
	categoryid INTEGER,

	quantity INTEGER,
	suppliercode VARCHAR(25),
	imageURL VARCHAR(255),
	active BOOLEAN,
	instock BOOLEAN,
	
	CONSTRAINT pk_product_id PRIMARY KEY (id)
);
